// Cola Simple Lineal Est√°tica

int frente = -1, final = -1, dato, i, opcion, max = 5;
int[] cola = new int[5];

‚ûï Encolar

case 1: // Encolar
    if (final == max - 1)
    {
        Console.WriteLine("Cola llena");
    }
    else
    {
        Console.Write("Ingrese dato: ");
        dato = int.Parse(Console.ReadLine());
        
        final++;
        cola[final] = dato;
        
        if (frente == -1)
        {
            frente = 0;
        }
        Console.WriteLine("Dato encolado en posici√≥n " + final);
    }
    break;

‚ûñ Desencolar

case 2: // Desencolar
    if (frente == -1 || frente > final)
    {
        Console.WriteLine("Cola vac√≠a, no se puede eliminar");
    }
    else
    {
        Console.WriteLine("Dato desencolado: " + cola[frente]);
        cola[frente] = 0;
        
        if (frente == final)
        {
            frente = -1;
            final = -1;
        }
        else
        {
            frente++;
        }
    }
    break;

üîç Consultar

case 3: // Consultar
    if (frente == -1)
    {
        Console.WriteLine("Cola vac√≠a");
    }
    else
    {
        Console.Write("Ingrese dato a buscar: ");
        dato = int.Parse(Console.ReadLine());
        bool encontrado = false;
        
        for (i = frente; i <= final; i++)
        {
            if (cola[i] == dato)
            {
                Console.WriteLine("Dato encontrado en posici√≥n " + i);
                encontrado = true;
                break;
            }
        }
        
        if (!encontrado)
        {
            Console.WriteLine("Dato no encontrado");
        }
    }
    break;
