// Cola 1: crece hacia la derecha (‚Üí) desde posici√≥n 0
// Cola 2: crece hacia la izquierda (‚Üê) desde posici√≥n max-1

int frente1 = -1, final1 = -1;  // Cola 1 (‚Üí)
int frente2 = -1, final2 = -1;  // Cola 2 (‚Üê)
int dato, i, opcion, max = 10;
int[] cola = new int[10];

‚û°Ô∏è Encolar desde el Frente (Cola 1)

case 1: // Encolar desde el frente - Cola 1 (crece ‚Üí)
    if (final1 + 1 == final2 || (final2 == -1 && final1 == max - 1))
    {
        Console.WriteLine("No hay espacio para encolar a la derecha");
    }
    else
    {
        Console.Write("Ingrese dato: ");
        dato = int.Parse(Console.ReadLine());
        final1++;
        cola[final1] = dato;
        
        if (frente1 == -1)
        {
            frente1 = 0;
        }
        Console.WriteLine("Dato encolado en posici√≥n " + final1);
    }
    break;

‚¨ÖÔ∏è Encolar desde Atr√°s (Cola 2)

case 2: // Encolar desde atr√°s - Cola 2 (crece ‚Üê)
    if (final2 - 1 == final1 || (final1 == -1 && final2 == 0))
    {
        Console.WriteLine("No hay espacio para encolar a la izquierda");
    }
    else
    {
        Console.Write("Ingrese dato: ");
        dato = int.Parse(Console.ReadLine());
        
        if (final2 == -1)
        {
            final2 = max - 1;
            frente2 = max - 1;
        }
        else
        {
            final2--;
        }
        cola[final2] = dato;
        Console.WriteLine("Dato encolado en posici√≥n " + final2);
    }
    break;

‚ûñ Desencolar Cola 1 (izquierda)

case 3: // Desencolar Cola 1

    {

        }
    }

case 5: // Buscar en ambas colas
    Console.Write("Ingrese dato a buscar: ");
    dato = int.Parse(Console.ReadLine());
    bool encontrado = false;
    
    // Buscar en Cola 1 (frente1 a final1)
    if (frente1 != -1)
    {
        for (i = frente1; i <= final1; i++)
        {
            if (cola[i] == dato)
            {
                Console.WriteLine("Encontrado en Cola 1, posici√≥n " + i);
                encontrado = true;
                break;
            }
        }
    }
    
    // Buscar en Cola 2 (frente2 a final2, decreciente)
    if (!encontrado && frente2 != -1)
    {
        for (i = frente2; i >= final2; i--)
        {
            if (cola[i] == dato)
            {
                Console.WriteLine("Encontrado en Cola 2, posici√≥n " + i);
                encontrado = true;
                break;
            }
        }
    }
    
    if (!encontrado)
    {
        Console.WriteLine("Dato no encontrado en ninguna cola");
    }
    break;
